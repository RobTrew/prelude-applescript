-- deleteFirst :: a -> [a] -> [a]
on deleteFirst(x, xs)
    script go
        on |位|(xs)
            if 0 < length of xs then
                tell xs to set {h, t} to {item 1, rest}
                if x = h then
                    t
                else
                    {h} & |位|(t)
                end if
            else
                {}
            end if
        end |位|
    end script
    go's |位|(xs)
end deleteFirst
