-- permutations :: [a] -> [[a]]
on permutations(xs)
	script firstElement
		on |位|(x)
			script tailElements
				on |位|(ys)
					{{x} & ys}
				end |位|
			end script
			
			concatMap(tailElements, permutations(|delete|(x, xs)))
		end |位|
	end script
	
	if length of xs > 0 then
		concatMap(firstElement, xs)
	else
		{{}}
	end if
end permutations
